form-colorchannel.ts ts 

import {
	asInteger,
	asNumber,
	type Component,
	component,
	on,
	setProperty,
} from '../../..'

export type FormColorchannelAxis = 'l' | 'c' | 'h'

export type FormColorchannelProps = {
	value: number
	min: number
	max: number
	stepDown: (shiftKey?: boolean) => void
	stepUp: (shiftKey?: boolean) => void
}

// const AXIS_MAX = { l: 1, c: 0.4, h: 360 }
// const AXIS_STEP = { l: 0.0025, c: 0.001, h: 1 }
// const AXIS_BIGSTEP = { l: 0.05, c: 0.02, h: 15 }

export default component<FormColorchannelProps>(
	'form-colorchannel',
	{
		value: 0,
		min: asNumber(),
		max: asNumber(1),
		stepDown: () => {},
		stepUp: () => {},
	},
	(el, { first }) => {
		const step = asNumber(1)(el, el.getAttribute('step'))
		const bigstep = asNumber(10)(el, el.getAttribute('bigstep'))
		const getStep = (shiftKey: boolean) => (shiftKey ? bigstep : step)
		const digits = asInteger(2)(el, el.getAttribute('digits'))
		const formatNumber = new Intl.NumberFormat('en-US', {
			maximumFractionDigits: digits,
		}).format

		// Step methods
		const setToNearestStep = (value: number) => {
			const nearest = Math.round(value / step) * step
			if (nearest >= el.min && nearest <= el.max) el.value = nearest
		}
		el.stepDown = (shiftKey = false) => {
			setToNearestStep(el.value - getStep(shiftKey))
		}
		el.stepUp = (shiftKey = false) => {
			setToNearestStep(el.value + getStep(shiftKey))
		}

		// Effects
		const effects = [
			first(
				'input',
				[
					setProperty('value', () => formatNumber(el.value)),
					on('change', ({ target }) => {
						el.value = target.valueAsNumber
					}),
				],
				'Add an <input[type="number"]> element to control the color channel.',
			),
			first(
				'button.decrement',
				[
					on('click', ({ event }) => {
						el.stepDown(event.shiftKey)
					}),
					setProperty('disabled', () => el.value <= el.min),
				],
				'Add a <button.decrement> to decrement a value for the color channel.',
			),
			first(
				'button.increment',
				[
					on('click', ({ event }) => {
						el.stepUp(event.shiftKey)
					}),
					setProperty('disabled', () => el.value >= el.max),
				],
				'Add a <button.increment> to increment a value for the color channel.',
			),
			on('keydown', ({ event }) => {
				const { key, shiftKey } = event
				const target = event.target as HTMLElement | null
				if (
					!target ||
					(target.localName === 'input' &&
						(key === 'ArrowLeft' || key === 'ArrowRight'))
				)
					return
				if (
					key.substring(0, 5) === 'Arrow' ||
					['+', '-'].includes(key)
				) {
					event.preventDefault()
					event.stopPropagation()
					if (
						key === 'ArrowLeft' ||
						key === 'ArrowDown' ||
						key === '-'
					)
						el.stepDown(shiftKey)
					else if (
						key === 'ArrowRight' ||
						key === 'ArrowUp' ||
						key === '+'
					)
						el.stepUp(shiftKey)
				}
			}),
		]
		return effects
	},
)

declare global {
	interface HTMLElementTagNameMap {
		'form-colorchannel': Component<FormColorchannelProps>
	}
}
